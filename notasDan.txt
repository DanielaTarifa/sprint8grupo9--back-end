listar:(req,res)=>{
        db.Users.findAll()
        .then(listarUser => {
            
            

            let respuesta={
                meta: {
                    status :200,
                    count:listarUser.length,
                },
                data:{list:[]}
                
            }
            
            listarUser.forEach(user=>{
                respuesta.data.list.push({
                    id:user.id,
                    name:user.name,
                    userName:user.userName,
                    email:user.email,
                    detail:`api/listarUser`
                })
                return user
            })

            return res.status(200).json(respuesta)
        })
        .catch(error => {res.send({error:'Not found'});})
        
    },
